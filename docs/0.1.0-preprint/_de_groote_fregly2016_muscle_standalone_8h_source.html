127  T calcActiveForceLengthMultiplier(const T& normFiberLength) const {
128  static const double b11 = 0.815;
129  static const double b21 = 1.055;
130  static const double b31 = 0.162;
131  static const double b41 = 0.063;
132  static const double b12 = 0.433;
133  static const double b22 = 0.717;
134  static const double b32 = -0.030;
135  static const double b42 = 0.200;
136  static const double b13 = 0.100;
137  static const double b23 = 1.000;
138  static const double b33 = 0.354;
139  static const double b43 = 0.000;
140  // Sum of 3 gaussians.
141  return gaussian(normFiberLength, b11, b21, b31, b41) +
142  gaussian(normFiberLength, b12, b22, b32, b42) +
143  gaussian(normFiberLength, b13, b23, b33, b43);

145  T calcForceVelocityMultiplier(const T& normFiberVelocity) const {
146 
147  // Muscle force-velocity.
148  static const double d1 = -0.318;
149  static const double d2 = -8.149;
150  static const double d3 = -0.374;
151  static const double d4 = 0.886;
152 
153  const T tempV = d2*normFiberVelocity + d3;
154  const T tempLogArg = tempV + sqrt(pow(tempV, 2) + 1);
155  return d1*log(tempLogArg) + d4;
156  }
157  T calcNormPassiveFiberForce(const T& normFiberLength) const {
158  // Passive force-length curve.
159  static const double kPE = 4.0;
160  static const double e0 = 0.6;
161  static const double denom = exp(kPE) - 1;
162  static const double numer_offset = exp(kPE * (0.2 - 1)/e0);
163  // The version of this equation in the supplementary materials of De
164  // Groote, et al. 2016 has an error. The correct equation passes
165  // through y = 0 at x = 0.2, and therefore is never negative within
166  // the allowed range of the optimal fiber length. The version in the
167  // supplementary materials allows for negative forces.
168  return (exp(kPE*(normFiberLength - 1)/e0) - numer_offset) / denom;
169  }
170  T calcNormFiberForce(const T& activation,
171  const T& normFiberLength,
172  const T& normFiberVelocity) const {
173 
174  // Active force-length curve.
175  const T activeForceLenMult =
176  calcActiveForceLengthMultiplier(normFiberLength);
177 
178  // Passive force-length curve.
179  const T normPassiveFibForce =
180  calcNormPassiveFiberForce(normFiberLength);
181 
182  // Force-velocity curve.
183  const T forceVelMult = calcForceVelocityMultiplier(normFiberVelocity);
184 
185  //std::cout << "DEBUG fib force comps "
186  // << static_cast<const double&>(activation + 0) << " "
187  // << static_cast<const double&>(activeForceLenMult + 0) << " "
188  // << static_cast<const double&>(forceVelMult + 0) << " "
189  // << static_cast<const double&>(normPassiveFibForce + 0) << " "
190  // << std::endl;
191  return activation*activeForceLenMult*forceVelMult + normPassiveFibForce;
192  }
193  T calcNormFiberForceAlongTendon(const T& activation,
194  const T& normFiberLength,

207  void calcRigidTendonFiberKinematics(const S& musTenLength,
208  const S& musTenVelocity,
209  S& normFiberLength,
210  S& normFiberVelocity) const {
211  // TODO there is too much repetition of calculations surrounding
212  // lengths and pennation.
213  // TODO can we use temporaries (adub etc) to speed up calculation?
214  const S fiberLengthAlongTendon = musTenLength - _tendon_slack_length;
215  const S fiberLength = sqrt(pow(musTenLength - _tendon_slack_length, 2)
216  + pow(_fiber_width, 2));
217  normFiberLength = fiberLength / _optimal_fiber_length;
218  const S cosPenn = fiberLengthAlongTendon / fiberLength;
219  // lMT = lT + lM cos(alpha) -> differentiate:
220  // vMT = vM cos(alpha) - lM alphaDot sin(alpha) (1)
221  // w = lM sin(alpha) -> differentiate:
222  // 0 = lMdot sin(alpha) + lM alphaDot cos(alpha) ->
223  // alphaDot = -vM sin(alpha) / (lM cos(alpha)) -> plug into (1)
224  // vMT = vM cos(alpha) + vM sin^2(alpha) / cos(alpha)
225  // vMT = vM (cos^2(alpha) + sin^2(alpha)) / cos(alpha)
226  // vM = vMT cos(alpha)
227  const S fiberVelocity = musTenVelocity * cosPenn;
228  // TODO cache max_contraction_velocity * opt_fib_len
229  normFiberVelocity = fiberVelocity /
230  (_max_contraction_velocity * _optimal_fiber_length);
231  }
234  T calcRigidTendonFiberForceAlongTendon(const T& activation,
235  const T& musTenLength,
236  const T& musTenVelocity) const {
237  T normFiberLength;
238  T normFiberVelocity;
239  calcRigidTendonFiberKinematics(musTenLength, musTenVelocity,
240  normFiberLength, normFiberVelocity);
241  // TODO what about buckling the tendon (MTU length < slack length)?
242  const T normFiberForce = calcNormFiberForceAlongTendon(
243  activation, normFiberLength, normFiberVelocity);
244  return _max_isometric_force * normFiberForce;
245  }
246  void calcActivationDynamics(const T& excitation, const T& activation,
247  T& activationDot) const {
248  static const double actTimeConst = 0.015;
249  static const double deactTimeConst = 0.060;
250  static const double tanhSteepness = 0.1;
251  // f = 0.5 tanh(b(e - a))
252  // z = 0.5 + 1.5a
253  // da/dt = [(f + 0.5)/(tau_a * z) + (-f + 0.5)*z/tau_d] * (e - a)
254  const T timeConstFactor = 0.5 + 1.5 * activation;
255  const T tempAct = 1.0 / (actTimeConst * timeConstFactor);
256  const T tempDeact = timeConstFactor / deactTimeConst;
257  const T f = 0.5 * tanh(tanhSteepness * (excitation - activation));
258  const T timeConst = tempAct * (f + 0.5) + tempDeact * (-f + 0.5);
259  activationDot = timeConst * (excitation - activation);
260  }

291  void calcIntermediatesForTendonForceState(const T& musTenLength,
292  const T& musTenVelocity, const T& normTenForce,
293  const T& normTenForceRate,
294  T& normFibLength, T& normFibVelocity, T& cosPenn, T& tendonForce)
295  const {

303  cosPenn = (musTenLength - tendonLength) / fiberLength;
304

306  calcNormTendonVelocity(normTenForceRate, normTenLength);
