44 template<typename T>
45 class INDYGOProblemInternal : public tropter::Problem<T> {
46 public:
47  INDYGOProblemInternal(const INDYGO& mrs,
48  const Model& model,
49  const InverseMuscleSolverMotionData& motionData,
50  const std::string& fiberDynamicsMode)
51  : tropter::Problem<T>("INDYGO"),
52  _mrs(mrs), _model(model), _motionData(motionData),
53  _useFiberLengthState(fiberDynamicsMode == "fiber_length") {
54  SimTK::State state = _model.initSystem();
55 
56  // Set the time bounds.
57  _initialTime = motionData.getInitialTime();
58  _finalTime = motionData.getFinalTime();
59  this->set_time({_initialTime}, {_finalTime});
60 
61  // States and controls for actuators.
62  // ----------------------------------
63  // TODO handle different actuator types more systematically.
64 
65  // CoordinateActuators.
66  // --------------------
67  _numCoordActuators = 0;
68  for (const auto& actuator :
69  _model.getComponentList<CoordinateActuator>()) {
70  if (!actuator.get_appliesForce()) continue;
71 
72  const auto& actuPath = actuator.getAbsolutePathString();
73  this->add_control(actuPath + "_control",
74  {actuator.get_min_control(),
75  actuator.get_max_control()});
76 
77  _otherControlsLabels.push_back(actuPath);
78  _numCoordActuators++;
79  }
80  const auto& coordPathsToActuate = motionData.getCoordinatesToActuate();
81  _optimalForce.resize(_numCoordActuators);
82  _coordActuatorDOFs.resize(_numCoordActuators);
83  const auto modelPath = model.getAbsolutePath();
84  int i_act = 0;
85  for (const auto& actuator :
86  _model.getComponentList<CoordinateActuator>()) {
87  if (!actuator.get_appliesForce()) continue;
88  _optimalForce[i_act] = actuator.getOptimalForce();
89  // Figure out which DOF each coordinate actuator is actuating.
90  const auto* coord = actuator.getCoordinate();
91  const auto coordPath = coord->getAbsolutePathString();
92  size_t i_coord = 0;
93  while (i_coord < coordPathsToActuate.size() &&
94  coordPathsToActuate[i_coord] != coordPath) {
95  ++i_coord;
96  }
97  // TODO move this into InverseMuscleSolver.
98  if (i_coord == coordPathsToActuate.size()) {
99  throw std::runtime_error("[INDYGO] Could not find Coordinate '" +
100  coord->getAbsolutePathString() + "' used in "
101  "CoordinateActuator '" +
102  actuator.getAbsolutePathString()
103  + "'. Is the coordinate locked?");
104  }
105  _coordActuatorDOFs[i_act] = i_coord;
106  ++i_act;
107  }
108 
109  // Muscles.
110  // --------
111  _numMuscles = 0;
112  for (const auto& actuator : _model.getComponentList<Muscle>()) {
113  // TODO consolidate with MotionData.
114  if (!actuator.get_appliesForce()) continue;
115 
116  const auto& actuPath = actuator.getAbsolutePathString();
117  // Activation dynamics.
118  // PR #1728 on opensim-core causes the min_control for muscles
119  // to be the minimum activation, and using non-zero minimum
120  // excitation/activation here causes issues (very noisy
121  // excitation/activation). Moreover, allowing zero muscle
122  // activation is possible with implicit muscle models and has
123  // numerical benefits.
124  this->add_control(actuPath + "_excitation", {0, 1});
125  // {actuator.get_min_control(),
126  // actuator.get_max_control()});
127  // TODO use activation bounds, not excitation bounds (then
128  // also update the property comment for zero_initial_activation).
129  const double initialActiv = mrs.get_zero_initial_activation() ?
130  0.0 : SimTK::NaN;
131  // const double initialActiv = mrs.get_zero_initial_activation() ?
132  // std::max(actuator.get_min_control(), 0.0) : SimTK::NaN;
133  this->add_state(actuPath + "_activation", {0, 1},
134  initialActiv); // TODO fix noisy activation bug
135  // {actuator.get_min_control(),
136  // actuator.get_max_control()},
137 
138  // Fiber dynamics.
139  if (_useFiberLengthState) {
140  // TODO initial value should be 0? That is what CMC does (via
141  // equilibrateMuscles()).
142  this->add_control(actuPath + "_norm_fiber_velocity", {-1, 1});
143  // These bounds come from simtk.org/projects/optcntrlmuscle.
144  this->add_state(actuPath + "_norm_fiber_length", {0.2, 1.8});
145  } else {
146  // TODO play with these bounds.
147  this->add_control(actuPath + "_tendon_force_rate_control",
148  {-50, 50});
149  this->add_state(actuPath + "_norm_tendon_force", {0, 5});
150  }
151  this->add_path_constraint(actuPath + "_equilibrium", 0);
152 
153  _muscleLabels.push_back(actuPath);
154  _numMuscles++;
155  }
156 
157  // Create De Groote muscles.
158  _muscles.resize(_numMuscles);
159  int i_mus = 0;
160  for (const auto& osimMus : _model.getComponentList<Muscle>()) {
161  if (!osimMus.get_appliesForce()) continue;
162 
163  _muscles[i_mus] = DeGrooteFregly2016MuscleStandalone<T>(
164  osimMus.get_max_isometric_force(),
165  osimMus.get_optimal_fiber_length(),
166  osimMus.get_tendon_slack_length(),
167  osimMus.get_pennation_angle_at_optimal(),
168  osimMus.get_max_contraction_velocity());
169 
170  i_mus++;
171  }
172 
173  this->add_cost("effort", 1);
174 
175  // Add a constraint for each coordinate we want to actuate.
176  _numCoordsToActuate = (int)coordPathsToActuate.size();
177  for (const auto& coordPath : coordPathsToActuate) {
178  this->add_path_constraint("net_gen_force_" + coordPath, 0);
179  }
180  }
181 
182  void initialize_on_mesh(const Eigen::VectorXd& mesh) const override {
183 
184  // For caching desired joint moments.
185  auto* mutableThis = const_cast<INDYGOProblemInternal<T>*>(this);
186 
187  // "array()" b/c Eigen matrix types do not support elementwise add.
188  // Writing the equation this way (rather than tau*(t_f - t_i) + t_i)
189  // avoids issues with roundoff where times.tail(1) != _finalTime.
190  const auto meshArray = mesh.array();
191  Eigen::VectorXd times =
192  (1 - meshArray) * _initialTime + meshArray * _finalTime;
193 
194  _motionData.interpolateNetGeneralizedForces(times,
195  mutableThis->_desiredMoments);
196  if (_numMuscles) {
197  _motionData.interpolateMuscleTendonLengths(times,
198  mutableThis->_muscleTendonLengths);
199  _motionData.interpolateMomentArms(times,
200  mutableThis->_momentArms);
201  if (!_useFiberLengthState) {
202  _motionData.interpolateMuscleTendonVelocities(times,
203  mutableThis->_muscleTendonVelocities);
204  }
205  }
206  }

208  void calc_differential_algebraic_equations(
209  const tropter::Input<T>& in,
210  tropter::Output<T> out) const override {
211 
212  const auto& i_time = in.time_index;
213  const auto& states = in.states;
214  const auto& controls = in.controls;
215 
216  // Actuator dynamics.
217  // ==================
218  for (Eigen::Index i_act = 0; i_act < _numMuscles; ++i_act) {
219 
220  // Unpack variables.
221  const T& excitation = controls[_numCoordActuators + 2 * i_act];
222  const T& activation = states[2 * i_act];
223 
224  // Activation dynamics.
225  _muscles[i_act].calcActivationDynamics(excitation, activation,
226  out.dynamics[2 * i_act]);
227 
228  // Fiber dynamics.
229  if (_useFiberLengthState) {
230  const T& normFibVel =
231  controls[_numCoordActuators + 2 * i_act + 1];
232  out.dynamics[2 * i_act + 1] =
233  _muscles[i_act].get_max_contraction_velocity()
234  * normFibVel;
235  } else {
236  const T& tenForceRateControl =
237  controls[_numCoordActuators + 2 * i_act + 1];
238  out.dynamics[2 * i_act + 1] = _tendonForceDynamicsScalingFactor
239  * tenForceRateControl;
240  }
241  }
242 
243  // TODO std::cout << "DEBUG dynamics " << derivatives << std::endl;
244 
245  // Actuator equilibrium.
246  // =====================
247 
248  // Assemble generalized forces to apply to the joints.
249  tropter::VectorX<T> genForce(_numCoordsToActuate);
250  genForce.setZero();
251 
252  // CoordinateActuators.
253  // --------------------
254  for (Eigen::Index i_act = 0; i_act < _numCoordActuators; ++i_act) {
255  genForce[_coordActuatorDOFs[i_act]]
256  += _optimalForce[i_act] * controls[i_act];
257  }
258 
259  // Muscles.
260  // --------
261  if (_numMuscles) {
262  tropter::VectorX<T> tendonForces(_numMuscles);
263  for (Eigen::Index i_act = 0; i_act < _numMuscles; ++i_act) {
264  // Unpack variables.
265  const T& activation = states[2 * i_act];
266  // Get the total muscle-tendon length from the data.
267  const T& musTenLen = _muscleTendonLengths(i_act, i_time);
268 
269  if (_useFiberLengthState) {
270  const T& normFibVel = controls[_numCoordActuators+2*i_act+1];
271  const T& normFibLen = states[2 * i_act + 1];
272  T normTenForce;
273  _muscles[i_act].calcEquilibriumResidual(activation,
274  musTenLen, normFibLen, normFibVel,
275  out.path[i_act],
276  normTenForce);
277  tendonForces[i_act] =
278  _muscles[i_act].get_max_isometric_force()
279  * normTenForce;
280  } else {
281  const T& tenForceRateControl =
282  controls[_numCoordActuators + 2 * i_act + 1];
283  const T& normTenForce = states[2 * i_act + 1];
284  const T& musTenVel = _muscleTendonVelocities(i_act, i_time);
285 
286  const T normTenForceRate =
287  _tendonForceDynamicsScalingFactor
288  * tenForceRateControl;
289  _muscles[i_act].calcTendonForceStateEquilibriumResidual(
290  activation, musTenLen, musTenVel, normTenForce,
291  normTenForceRate,
292  out.path[i_act], tendonForces[i_act]);
293  }
294  }
295 
296  // Compute generalized forces from muscles.
297  const auto& momArms = _momentArms[i_time];
298  // TODO convert to type T once instead of in every iteration.
299  genForce += momArms.template cast<T>() * tendonForces;
300  }
301 
302 
303  // Achieve the motion.
304  // ===================
305  if (out.path.size() != 0) {
306  out.path.segment(_numMuscles, _numCoordsToActuate)
307  = _desiredMoments.col(i_time).template cast<T>()
308  - genForce;
309  }
310  }
311  void calc_cost(int cost_index, const tropter::CostInput<T>& in,
312  T& cost) const override {
313  cost = in.integral;
314  }
315  void calc_cost_integrand(int cost_index, const tropter::Input<T>& in,
316  T& integrand) const override {
317  // Unpack variables.
318  const auto& states = in.states;
319  const auto& controls = in.controls;
320  // Use a map to skip over fiber velocities.
321  using ExcitationsVector = Eigen::Map<const tropter::VectorX<T>,
322  /* pointer alignment: Unaligned */ 0,
323  /* pointer increment btn elements */ Eigen::InnerStride<2>>;
324  ExcitationsVector muscleExcit(controls.data() + _numCoordActuators,
325  _numMuscles);
326 
327  // Minimize activations to prevent the initial
328  // activation from being incorrectly large (b/c no penalty)
329  // TODO could also just minimize initial activation.
330  // Use a map to skip over fiber lengths.
331  using ActivationsVector = Eigen::Map<const tropter::VectorX<T>,
332  /* pointer alignment: Unaligned */ 0,
333  /* pointer increment btn elements */ Eigen::InnerStride<2>>;
334  ActivationsVector muscleActiv(states.data(), _numMuscles);
335 
336  integrand = controls.head(_numCoordActuators).squaredNorm()
337  + muscleExcit.squaredNorm() +
338  + muscleActiv.squaredNorm();
339  }
343  // TODO should take a "Iterate" instead.
344  tropter::Iterate construct_iterate(
345  const INDYGO::Solution& mrsVars) const {
346  using Eigen::Map;
347  using Eigen::VectorXd;
348  using Eigen::MatrixXd;
349 
350  tropter::Iterate vars;
351  // The mrsVars has time as the row dimension, but mrsVars has time as
352  // the column dimension. As a result, some quantities must be
353  // transposed.
354  size_t numTimes;
355  if (mrsVars.activation.getNumRows() != 0) {
356  numTimes = mrsVars.activation.getNumRows();
357  vars.time = Map<const VectorXd>(
358  mrsVars.activation.getIndependentColumn().data(), numTimes);
359  } else if (mrsVars.other_controls.getNumRows() != 0) {
360  numTimes = mrsVars.other_controls.getNumRows();
361  vars.time = Map<const VectorXd>(
362  mrsVars.other_controls.getIndependentColumn().data(),
363  numTimes);
364  } else {
365  // mrsVars seems to be empty (no muscles or other controls);
366  // return an empty Iterate.
367  return vars;
368  }
369 
370  // Allocate memory.
371  // Each muscle has excitation and norm. fiber velocity (or tendon force
372  // rate) controls.
373  const int numControls = _numCoordActuators + 2 * _numMuscles;
374  // Each muscle has activation and fiber length (or tendon force) states.
375  const int numStates = 2 * _numMuscles;
376  vars.controls.resize(numControls, numTimes);
377  vars.states.resize(numStates, numTimes);
378  vars.control_names.resize(numControls);
379  vars.state_names.resize(2 * _numMuscles);
380 
381  if (_numCoordActuators) {
382  // Set control names.
383  const auto& otherControlsLabels =
384  mrsVars.other_controls.getColumnLabels();
385  assert(_numCoordActuators == (int)otherControlsLabels.size());
386  for (int i_act = 0; i_act < _numCoordActuators; ++i_act) {
387  vars.control_names[i_act] =
388  otherControlsLabels[i_act] + "_control";
389  }
390  vars.controls.topRows(_numCoordActuators) = Map<const MatrixXd>(
391  &mrsVars.other_controls.getMatrix()(0, 0),
392  numTimes, _numCoordActuators).transpose();
393  }
394 
395  if (_numMuscles) {
396  // Set state and control names.
397  const auto& muscleNames = mrsVars.activation.getColumnLabels();
398  for (int i_musc = 0; i_musc < _numMuscles; ++i_musc) {
399  const auto& muscleName = muscleNames[i_musc];
400  vars.control_names[_numCoordActuators + 2 * i_musc] =
401  muscleName + "_excitation";
402  vars.control_names[_numCoordActuators + 2 * i_musc + 1] =
403  muscleName +
404  (_useFiberLengthState ? "_norm_fiber_velocity"
405  : "_tendon_force_rate_control");
406  vars.state_names[2 * i_musc] = muscleName + "_activation";
407  vars.state_names[2 * i_musc + 1] = muscleName +
408  (_useFiberLengthState ? "_norm_fiber_length"
409  : "_norm_tendon_force");
410  }
411  // SkipRowView is a view onto a Matrix wherein every other row of
412  // the original matrix is skipped. We use this to edit every
413  // other row using a single assignment. Eigen's inner stride (the
414  // second template argument below) is the pointer increment between
415  // two consecutive entries; so a stride of 2 skips 1 row.
416  // The first argument is the outer stride (increment between
417  // columns), which we only know during run-time (e.g., Dynamic).
418  using SkipRowStride = Eigen::Stride<Eigen::Dynamic, 2>;
419  // These tell Map to interpret the passed-in data as a matrix;
420  // the outer stride is just the number of rows (in this case).
421  SkipRowStride controlsStride(vars.controls.outerStride(), 2);
422  SkipRowStride statesStride(vars.states.outerStride(), 2);
423  using SkipRowView = Map<MatrixXd, Eigen::Unaligned, SkipRowStride>;
424 
425  SkipRowView excitationView(
426  // Skip over coordinate actuator controls.
427  vars.controls.bottomRows(2 *_numMuscles).data(),
428  _numMuscles, numTimes, controlsStride);
429  excitationView = Map<const MatrixXd>(
430  &mrsVars.excitation.getMatrix()(0, 0),
431  numTimes, _numMuscles).transpose();
432 
433  SkipRowView activationView(
434  vars.states.data(), _numMuscles, numTimes, statesStride);
435  activationView = Map<const MatrixXd>(
436  &mrsVars.activation.getMatrix()(0, 0),
437  numTimes, _numMuscles).transpose();
438 
439  if (_useFiberLengthState) {
440  SkipRowView fiberLengthView(
441  // Skip the first state row, which has activations
442  // for the first muscle.
443  vars.states.bottomRows(2 * _numMuscles - 1).data(),
444  _numMuscles, numTimes, statesStride);
445  fiberLengthView = Map<const MatrixXd>(
446  &mrsVars.norm_fiber_length.getMatrix()(0, 0),
447  numTimes, _numMuscles).transpose();
448 
449  SkipRowView fiberVelocityView(
450  // Skip the coordinate actuator rows and the excitation
451  // for the first muscle.
452  vars.controls.bottomRows(2 * _numMuscles - 1).data(),
453  _numMuscles, numTimes, controlsStride);
454  fiberVelocityView = Map<const MatrixXd>(
455  &mrsVars.norm_fiber_velocity.getMatrix()(0, 0),
456  numTimes, _numMuscles).transpose();
457  } else {
458  SkipRowView tendonForceView(
459  vars.states.bottomRows(2 * _numMuscles - 1).data(),
460  _numMuscles, numTimes, statesStride);
461  tendonForceView = Map<const MatrixXd>(
462  &mrsVars.norm_tendon_force.getMatrix()(0, 0),
463  numTimes, _numMuscles).transpose();
464 
465  SkipRowView tendonForceRateControlView(
466  vars.controls.bottomRows(2 * _numMuscles - 1).data(),
467  _numMuscles, numTimes, controlsStride);
468  tendonForceRateControlView = Map<const MatrixXd>(
469  &mrsVars.tendon_force_rate_control.getMatrix()(0, 0),
470  numTimes, _numMuscles).transpose();
471  }
472  }
473  return vars;
474  }
479  // TODO rename to make_indygo_iterate and make_optimal_control_iterate
480  INDYGO::Solution deconstruct_iterate(
481  const tropter::Iterate& ocpVars) const {
482 
483  INDYGO::Solution sol;
484  if (_numCoordActuators) {
485  sol.other_controls.setColumnLabels(_otherControlsLabels);
486  }
487  if (_numMuscles) {
488  sol.excitation.setColumnLabels(_muscleLabels);
489  sol.activation.setColumnLabels(_muscleLabels);
490  sol.norm_fiber_length.setColumnLabels(_muscleLabels);
491  sol.norm_fiber_velocity.setColumnLabels(_muscleLabels);
492  sol.tendon_force.setColumnLabels(_muscleLabels);
493  sol.norm_tendon_force.setColumnLabels(_muscleLabels);
494  if (!_useFiberLengthState) {
495  sol.tendon_force_rate_control.setColumnLabels(_muscleLabels);
496  }
497  }
498 
499  for (int i_time = 0; i_time < ocpVars.time.cols(); ++i_time) {
500  const auto& time = ocpVars.time[i_time];
501  const auto& controls = ocpVars.controls.col(i_time);
502  const auto& states = ocpVars.states.col(i_time);
503 
504  // Other controls.
505  // ---------------
506  // The first _numCoordActuators rows of the controls matrix
507  // are for the CoordinateActuators.
508  if (_numCoordActuators) {
509  SimTK::RowVector other_controls(_numCoordActuators,
510  controls.data(),
511  true /* <- this is a view */);
512  sol.other_controls.appendRow(time, other_controls);
513  }
514 
515  // Muscle-related quantities.
516  // --------------------------
517  if (_numMuscles == 0) continue;
518  SimTK::RowVector excitation(_numMuscles,
519  2 /* stride: skip over fiber vel. */,
520  controls.data() + _numCoordActuators,
521  true /* makes this a view */);
522  sol.excitation.appendRow(time, excitation);
523 
524  SimTK::RowVector activation(_numMuscles,
525  2 /* stride: skip over fiber length */,
526  states.data(),
527  true /* makes this a view */);
528  sol.activation.appendRow(time, activation);
529 
530  if (_useFiberLengthState) {
531  SimTK::RowVector fiber_length(_numMuscles, /*TODO*/
532  2 /* stride: skip over activation */,
533  states.data() + 1,
534  true /* makes this a view */);
535  sol.norm_fiber_length.appendRow(time, fiber_length);
536 
537  SimTK::RowVector fiber_velocity(_numMuscles,
538  2 /* stride: skip over excit. */,
539  controls.data() + _numCoordActuators + 1,
540  true /* makes this a view */);
541  sol.norm_fiber_velocity.appendRow(time, fiber_velocity);
542 
543  // Compute other helpful quantities from the states.
544  SimTK::RowVector tendon_force(_numMuscles);
545  SimTK::RowVector norm_tendon_force(_numMuscles);
546  for (int i_musc = 0; i_musc < _numMuscles; ++i_musc) {
547  const auto muscle =
548  _muscles[i_musc].convert_scalartype_double();
549  const auto& musTenLen = _muscleTendonLengths(i_musc, i_time);
550  const auto& normFibLen = states[2 * i_musc + 1];
551 
552  norm_tendon_force[i_musc] =
553  muscle.calcNormTendonForce(musTenLen, normFibLen);
554  tendon_force[i_musc] = muscle.get_max_isometric_force() *
555  norm_tendon_force[i_musc];
556  }
557  sol.tendon_force.appendRow(time, tendon_force);
558  sol.norm_tendon_force.appendRow(time, norm_tendon_force);
559 
560  // Don't worry about tendon_force_rate_control. It's unlikely
561  // that users are interested in this. It will be necessary to
562  // carry over if one wants to first solve a problem with
563  // fiber length state and use the solution as an initial
564  // guess to a problem with tendon force state.
565  } else {
566  // TODO
567  SimTK::RowVector norm_tendon_force(_numMuscles,
568  2, states.data() + 1, true);
569  sol.norm_tendon_force.appendRow(time, norm_tendon_force);
570 
571  SimTK::RowVector tendon_force_rate_control(_numMuscles,
572  2, controls.data() + _numCoordActuators + 1,
573  true);
574  sol.tendon_force_rate_control.appendRow(time,
575  tendon_force_rate_control);
576 
577  // We must compute fiber length, fiber velocity, and tendon
578  // force from the states and controls.
579  SimTK::RowVector norm_fiber_length(_numMuscles);
580  SimTK::RowVector norm_fiber_velocity(_numMuscles);
581  SimTK::RowVector tendon_force(_numMuscles);
582  for (int i_musc = 0; i_musc < _numMuscles; ++i_musc) {
583  const auto muscle =
584  _muscles[i_musc].convert_scalartype_double();
585  const auto& musTenLen =
586  _muscleTendonLengths(i_musc, i_time);
587  const auto& musTenVel =
